Networkgraph chart (special Chart)

Networkgraph 차트를 표현합니다. modules/networkgraph.js 필요

Example

myChart.setOptions({
  chart: {
    type: 'networkgraph',
    
  },
  plotOptions: {
    sankey: {
		'keys': ['from','to'], // Networkgraph 에서 해당 코드 선언 필수
    }
  },
  series: [{
    // from 에서 to 로
    'data': [
      ['A', 'B'],
      ['A', 'C'],
      ['A', 'D'],
      ['A', 'E'],
      ['A', 'F'],
      ['A', 'G'],
      ['A', 'H'],
      ['A', 'I'],
      ['H', 'J'],
      ['H', 'K'],
      ['H', 'L'],
      ['L', 'M'],
      ['M', 'N'],
      ['K', 'O'],
      ['E', 'P'],
      ['E', 'Q'],
      ['Q', 'R'],
      ['D', 'S'],
      ['D', 'T']
    ]
}]
},
{
  append: true,
  redraw: true
});
