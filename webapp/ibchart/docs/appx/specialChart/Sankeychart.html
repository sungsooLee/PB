Sankey chart (special Chart)

Sankey 차트를 표현합니다. modules/sankey.js 필요

Example

myChart.setOptions({
  chart: {
    type: 'sankey',
    
  },
  plotOptions: {
    sankey: {
		'keys': ['from','to','weight'], // sankey 에서 해당 코드 선언 필수
    }
  },
  series: [{
      // from 에서 to 까지 weight 만큼
	'data': [
		["A","B",10],
		["A","C",2],
		["B","D",5],
		["B","F",3],
		["B","G",2],
		["C","D",1],
		["C","F",1]
	]
}]
},
{
  append: true,
  redraw: true
});
