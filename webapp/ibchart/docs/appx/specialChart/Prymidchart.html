Prymid chart (special Chart)

Prymid 차트를 표현합니다. highcharts-more.js 필요

Example

myChart.setOptions({
  chart:{
    type:'pyramid'
  },
  series: [{
    name: 'Unique users',
    data: [
      ['Website visits', 1565],
      ['Downloads', 4064],
      ['Requested price list', 1987],
      ['Invoice sent', 976],
      ['Finalized', 846]
    ]
  }]
},{
  append: true,
  redraw: true
});
