Angular gauges chart (special Chart)

gague 차트를 표현합니다. highcharts-more.js 필요

Example

myChart.setOptions({
  chart: {
    type: "gauge"
  },
  pane: {
    startAngle: -90, // 게이지 바 시작 각도
    endAngle: 90, // 게이지 바 종료 각도
  },
  yAxis: {
    min: 0, // 게이지 바 시작 값
    max: 200, // 게이지 바 종료 값
    tickInterval : 10 // 10마다 눈금 표시
 },
 series: [{
  data : [60]
}]
},{
  append: true,
  redraw: true
});
