plotLine 기능 (chartSkill)

차트 중간에 선을 표현 할 수 있습니다.

Example

myChart.setOptions({
  chart:{
    type:'column'
  },
  yAxis: {
    title:{
      text: "Y축 타이틀"
    },
    plotLines : [{
      color: '#FF0000',
      value:25,
      width:2,
      label:{
        text:"성장률평균",
        align:"right"
      },
      zIndex:10
    },{
      color: '#00FF00',
      value:481.5,
      width:2,
      label:{
        text:"판매량평균",
        align:"center"
      },
      zIndex:10
    }]
  },
  series: [{
    name : "성장률",
    zIndex:3,
    data : [10,32,15,53,18,22],
    dataLabels:{
      enabled:true
    }
  },{
    name : "판매량",
    zIndex:1,
    data : [580,100,700,400,600,509],
    dataLabels:{
      enabled:true
    }
  }]
},{
  append: true,
  redraw: true
});
