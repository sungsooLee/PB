파이 차트 크기 고정 (chartSkill)

파이 차트 크기 고정기능입니다. plotOptions.pie.size

Example

myChart.setOptions({
  chart:{
    type:'pie'
  },
  plotOptions: {
    pie: {
      dataLabels: {
        enabled: true,
        distance: -50
      },
      size: "100%"
    }
  },
  series: [{
    data : [
            ["축구", 20],
            ["농구", 40],
            ["배구", 30],
            ["야구", 45]
    ]
  }]
},{
  append: true,
  redraw: true
});
