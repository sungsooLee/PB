도넛 안에 타이틀 넣는 기능 (chartSkill)

도넛형 차트 내 타이틀을 넣을 수 있습니다.

Example

myChart.setOptions({
  colors: ['#DDDF00','#24CBE5','#64E572','#FF9655','#FFF263']
  chart:{
      type:'pie'
  },
  title:{
    text: "title",
    align: "center",
    verticalAlign: "middle"
  },
  plotOptions: {
    pie:{
      dataLabels: {
        enabled: true,
        distance: -50,
      }
    }
  },
  series: [{
    innerSize: "50%",
    data : [
      ["축구", 44]
      ["농구", 26],
      ["배구", 20],
      ["야구", 3],
    ]
  }]
},{
  append: true,
  redraw: true
});
