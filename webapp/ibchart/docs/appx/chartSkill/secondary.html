Secondary Y축 사용 (chartSkill)

두개의 Y축을 사용합니다.

Example

myChart.setOptions({
  chart:{
    type:'column'
  },
  yAxis: [{
    title:{
      text: "Y축 타이틀"
    }
  },{
    title:{
      text: "Y축 타이틀"
    },
    opposite: true
  }],
  series: [{
    name : "성장률",
    type:"line",
    zIndex:3,
    data : [10,32,15,53,18,22],
    dataLabels:{
      enabled:true
    }
  },{
    name : "판매량",
    zIndex:1,
    data : [580,100,700,400,600,509],
    dataLabels:{
      enabled:true
    },
    yAxis:1
  }]
},{
  append: true,
  redraw: true
});
