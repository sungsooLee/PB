type (chart properties)

name type required(default) description
type string 선택(line) 차트 유형

Example

//차트 기본 속성 설정
myChart.setOptions({
  chart: {
    type: 'column'  //세로 막대 차트
  },
  series: [{
    data : [60,40,20,80]
  }]
}, {
  append: true,
  redraw: true
});
